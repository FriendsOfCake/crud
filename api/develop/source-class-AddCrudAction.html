 30  */

 71 /**
 72  * HTTP POST handler
 73  *
 74  * @return void
 75  */
 76     protected function _post() {
 77         $request = $this->_request();
 78         $model = $this->_model();
 79 
 80         if ($request->data('_cancel')) {
 81             $subject = $this->_trigger('beforeCancel');
 82             $controller = $this->_controller();
 83             return $this->_redirect($subject, $controller->referer(array('action' => 'index')));
 84         }
 85 
 86         $this->_trigger('beforeSave');
 87         if ($model->saveAll($request->data, $this->saveOptions())) {
 88             $this->setFlash('success');
 89             $subject = $this->_trigger('afterSave', array('success' => true, 'created' => true, 'id' => $model->id));
 90 
 91             if ($request->data('_add')) {
 92                 return $this->_redirect($subject, array('action' => $request->action));
 93             } elseif ($request->data('_edit')) {
 94                 return $this->_redirect($subject, array('action' => 'edit', $model->id));
 95             }
 96 
 97             $controller = $this->_controller();
 98             return $this->_redirect($subject, $controller->referer(array('action' => 'index')));
 99         }
100 
101         $this->setFlash('error');
102         $this->_trigger('afterSave', array('success' => false, 'created' => false));
103         $request->data = Hash::merge($request->data, $model->data);
104         $this->_trigger('beforeRender', array('success' => false));
105     }
106
