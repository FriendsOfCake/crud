 56 /**
 57  * Sets a configuration variable into this action
 58  *
 59  * If called with no arguments, all configuration values are
 60  * returned.
 61  *
 62  * $key is interpreted with dot notation, like the one used for
 63  * Configure::write()
 64  *
 65  * If $key is string and $value is not passed, it will return the
 66  * value associated with such key.
 67  *
 68  * If $key is an array and $value is empty, then $key will
 69  * be interpreted as key => value dictionary of settings and
 70  * it will be merged directly with $this->settings
 71  *
 72  * If $key is a string, the value will be inserted in the specified
 73  * slot as indicated using the dot notation
 74  *
 75  * @param mixed $key
 76  * @param mixed $value
 77  * @param boolean $merge
 78  * @return mixed|CrudAction
 79  */
 80     public function config($key = null, $value = null, $merge = true) {
 81         if (is_null($key) && is_null($value)) {
 82             return $this->_settings;
 83         }
 84 
 85         if (is_null($value)) {
 86             if (is_array($key)) {
 87                 if ($merge) {
 88                     $this->_settings = Hash::merge($this->_settings, $key);
 89                 } else {
 90                     foreach (Hash::flatten($key) as $k => $v) {
 91                         $this->_settings = Hash::insert($this->_settings, $k, $v);
 92                     }
 93                 }
 94 
 95                 return $this;
 96             }
 97 
 98             return Hash::get($this->_settings, $key);
 99         }
100 
101         if (is_array($value)) {
102             if ($merge) {
103                 $value = array_merge((array)Hash::get($this->_settings, $key), $value);
104             } else {
105                 foreach ($value as $k => $v) {
106                     $this->_settings = Hash::insert($this->_settings, $k, $v);
107                 }
108             }
109         }
110 
111         $this->_settings = Hash::insert($this->_settings, $key, $value);
112         return $this;
113     }

259 /**
260  * Returns the redirect_url for this request.
261  *
262  * @param string $default Default URL to use redirect_url is not found in request or data
263  * @return mixed
264  */
265     protected function _redirectUrl($default = null) {
266         $url = $default;
267         $request = $this->_request();
268         if (!empty($request->data['redirect_url'])) {
269             $url = $request->data['redirect_url'];
270         } elseif (!empty($request->query['redirect_url'])) {
271             $url = $request->query['redirect_url'];
272         }
273 
274         return $url;
275     }
