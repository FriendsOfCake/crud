30 /**
31  * Appends the pagination information to the JSON or XML output
32  *
33  * @param CakeEvent $event
34  * @return void
35  */
36     public function beforeRender(CakeEvent $event) {
37         $request = $this->_request();
38         if (!$request->is('api')) {
39             return;
40         }
41 
42         $_pagination = $request->paging;
43         if (empty($_pagination) || !array_key_exists($event->subject->modelClass, $_pagination)) {
44             return;
45         }
46 
47         $_pagination = $_pagination[$event->subject->modelClass];
48 
49         $pagination = array(
50             'page_count' => $_pagination['pageCount'],
51             'current_page' => $_pagination['page'],
52             'has_next_page' => $_pagination['nextPage'],
53             'has_prev_page' => $_pagination['prevPage'],
54             'count' => $_pagination['count'],
55             'limit' => $_pagination['limit']
56         );
57 
58         $this->_action()->config('serialize.pagination', 'pagination');
59         $this->_controller()->set('pagination', $pagination);
60     }
