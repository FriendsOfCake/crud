 33 /**
 34  * Make sure to contain associated models
 35  *
 36  * This have no effect on clean applications where containable isn't
 37  * loaded, but for those who does have it loaded, we should
 38  * use it.
 39  *
 40  * This help applications with `$recursive -1` in their AppModel
 41  * and containable behavior loaded
 42  *
 43  * @param CakeEvent $event
 44  * @return void
 45  */
 46     public function beforeFind(CakeEvent $event) {
 47         if (!isset($event->subject->query['contain'])) {
 48             $event->subject->query['contain'] = array();
 49         }
 50 
 51         $existing = $event->subject->query['contain'];
 52         $associated = array_keys($event->subject->model->getAssociated());
 53 
 54         $event->subject->query['contain'] = array_merge($existing, $associated);
 55     }

 83 /**
 84  * Do all the magic needed for using the
 85  * cakephp scaffold views
 86  *
 87  * @param CakeEvent
 88  * @return void
 89  */
 90     public function beforeRender(CakeEvent $event) {
 91         $subject = $event->subject;
 92         $model = $subject->model;
 93         $request = $subject->request;
 94         $controller = $subject->controller;
 95 
 96         $scaffoldTitle = Inflector::humanize(Inflector::underscore($controller->viewPath));
 97         $title = __d('cake', 'Scaffold :: ') . Inflector::humanize($request->action) . ' :: ' . $scaffoldTitle;
 98 
 99         $modelClass = $controller->modelClass;
100         $primaryKey = $model->primaryKey;
101         $displayField = $model->displayField;
102         $singularVar = Inflector::variable($modelClass);
103         $pluralVar = Inflector::variable($controller->name);
104         $singularHumanName = Inflector::humanize(Inflector::underscore($modelClass));
105         $pluralHumanName = Inflector::humanize(Inflector::underscore($controller->name));
106         $scaffoldFields = array_keys($model->schema());
107         $associations = $this->_associations($model);
108 
109         $controller->set(compact(
110             'modelClass', 'primaryKey', 'displayField', 'singularVar', 'pluralVar',
111             'singularHumanName', 'pluralHumanName', 'scaffoldFields', 'associations'
112         ));
113 
114         $controller->set('title_for_layout', $title);
115 
116         if ($controller->viewClass) {
117             $controller->viewClass = 'Scaffold';
118         }
119     }

121 /**
122  * Returns associations for controllers models.
123  *
124  * @param Model $model
125  * @return array Associations for model
126  */
127     protected function _associations(Model $model) {
128         $associations = array();
129 
130         $associated = $model->getAssociated();
131         foreach ($associated as $assocKey => $type) {
132             if (!isset($associations[$type])) {
133                 $associations[$type] = array();
134             }
135 
136             $assocDataAll = $model->$type;
137 
138             $assocData = $assocDataAll[$assocKey];
139             $associatedModel = $model->{$assocKey};
140 
141             $associations[$type][$assocKey]['primaryKey'] = $associatedModel->primaryKey;
142             $associations[$type][$assocKey]['displayField'] = $associatedModel->displayField;
143             $associations[$type][$assocKey]['foreignKey'] = $assocData['foreignKey'];
144 
145             list($plugin, $modelClass) = pluginSplit($assocData['className']);
146 
147             if ($plugin) {
148                 $plugin = Inflector::underscore($plugin);
149             }
150 
151             $associations[$type][$assocKey]['plugin'] = $plugin;
152             $associations[$type][$assocKey]['controller'] = Inflector::pluralize(Inflector::underscore($modelClass));
153 
154             if ($type === 'hasAndBelongsToMany') {
155                 $associations[$type][$assocKey]['with'] = $assocData['with'];
156             }
157         }
158 
159         return $associations;
160     }
