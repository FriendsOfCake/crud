182  * Execute a Crud action
183  *
184  * @param string $controllerAction Override the controller action to execute as
185  * @param array $arguments List of arguments to pass to the CRUD action (Usually an ID to edit / delete)
186  * @return mixed void, or a CakeResponse object

189     public function executeAction($controllerAction = null, $args = array()) {
190         $this->_loadListeners();
191 
192         $this->_action = $controllerAction ?: $this->_action;
193 
194         $view = $action = $this->_action;
195         if (empty($args)) {

197         }
198 
199         try {
200             $subject = $this->trigger('beforeHandle', compact('args', 'action'));
201 
202             $response = $this->action($subject->action)->handle($subject);
203             if ($response instanceof CakeResponse) {
204                 return $response;
205             }
206         } catch (Exception $e) {
207             if (isset($e->response)) {
208                 return $e->response;
209             }
210

212         }

215         return $this->_controller->response = $this->_controller->render($view);

449 /**
450  * Triggers a Crud event by creating a new subject and filling it with $data
451  * if $data is an instance of CrudSubject it will be reused as the subject
452  * object for this event.
453  *
454  * If Event listeners return a CakeResponse object, the this method will throw an
455  * exception and fill a 'response' property on it with a reference to the response
456  * object.
457  *
458  * @param string $eventName
459  * @param array $data
460  * @throws Exception if any event listener return a CakeResponse object
461  * @return CrudSubject
462  */
463     public function trigger($eventName, $data = array()) {
464         $eventName = $this->settings['eventPrefix'] . '.' . $eventName;
465         $subject = $data instanceof CrudSubject ? $data : $this->getSubject($data);
466         $subject->addEvent($eventName);
467 
468         if (!empty($this->settings['eventLogging'])) {
469             $this->logEvent($eventName, $data);
470         }
471 
472         $event = new CakeEvent($eventName, $subject);
473         $this->_eventManager->dispatch($event);
474 
475         if ($event->result instanceof CakeResponse) {
476             $exception = new Exception();
477             $exception->response = $event->result;
478             throw $exception;
479         }
480 
481         $subject->stopped = false;
482         if ($event->isStopped()) {
483             $subject->stopped = true;
484         }
485 
486         return $subject;
487     }

503 /**
504  * Sets a configuration variable into this component
505  *
506  * If called with no arguments, all configuration values are
507  * returned.
508  *
509  * $key is interpreted with dot notation, like the one used for
510  * Configure::write()
511  *
512  * If $key is string and $value is not passed, it will return the
513  * value associated with such key.
514  *
515  * If $key is an array and $value is empty, then $key will
516  * be interpreted as key => value dictionary of settings and
517  * it will be merged directly with $this->settings
518  *
519  * If $key is a string, the value will be inserted in the specified
520  * slot as indicated using the dot notation
521  *
522  * @param mixed $key
523  * @param mixed $value
524  * @return mixed|CrudComponent
525  */
526     public function config($key = null, $value = null) {
527         if (is_null($key) && is_null($value)) {
528             return $this->settings;
529         }
530 
531         if (is_null($value)) {
532             if (is_array($key)) {
533                 $this->settings = Hash::merge($this->settings, $key);
534                 return $this;
535             }
536 
537             return Hash::get($this->settings, $key);
538         }
539 
540         if (is_array($value)) {
541             $value = array_merge((array)Hash::get($this->settings, $key), $value);
542         }
543 
544         $this->settings = Hash::insert($this->settings, $key, $value);
545         foreach (array('listeners', 'actions') as $type) {
546             if (strpos($key, $type . '.') === 0) {
547                 $this->_normalizeConfig($type);
548             }
549         }
550 
551         return $this;
552     }

649 /**
650  * Normalize action configuration
651  *
652  * If an action don't have a CrudClass specified (the value part of the array)
653  * try to compute it by exploding on action name on '_' and take the last chunk
654  * as CrudClass identifier
655  *
656  * @param mixed $types Class type(s)
657  * @return void
658  * @throws CakeException If className is missing for listener
659  */
660     protected function _normalizeConfig($types = null) {
661         if (!$types) {
662             $types = array('listeners', 'actions');
663         }
664 
665         foreach ((array)$types as $type) {
666             $this->settings[$type] = Hash::normalize($this->settings[$type]);
667 
668             foreach ($this->settings[$type] as $name => $settings) {
669                 if (is_array($settings) && !empty($settings['className'])) {
670                     $this->settings[$type][$name] = $settings;
671                     continue;
672                 }
673 
674                 $className = null;
675                 if (empty($settings)) {
676                     $settings = array();
677                 } elseif (is_string($settings)) {
678                     $className = $settings;
679                     $settings = array();
680                 }
681 
682                 if ($type === 'listeners' && false !== strpos($name, '.')) {
683                     unset($this->settings[$type][$name]);
684 
685                     list($plugin, $name) = pluginSplit($name);
686                     $name = Inflector::camelize($name);
687                 }
688 
689                 $className = $this->_handlerClassName($name, $className);
690                 $settings['className'] = $className;
691                 $this->settings[$type][$name] = $settings;
692             }
693         }
694     }

730  * Load a single event class attached to Crud
731  *
732  * @throws CakeException
733  * @param string $name
734  * @return void
735  */
736     protected function _loadListener($name) {
737         if (!isset($this->_listenerInstances[$name])) {
738             $config = $this->config('listeners.' . $name);
739 
740             if (empty($config)) {
741                 throw new CakeException(sprintf('Listener "%s" is not configured', $name));
742             }
743 
744             list($plugin, $class) = pluginSplit($config['className'], true);
745             $class .= 'Listener';
746             App::uses($class, $plugin . 'Controller/Crud/Listener');
747 
748             $subject = $this->getSubject();
749             $this->_listenerInstances[$name] = new $class($subject, $config);
750             $this->_eventManager->attach($this->_listenerInstances[$name]);
751             if (is_callable(array($this->_listenerInstances[$name], 'setup'))) {
752                 $this->_listenerInstances[$name]->setup();
753             }
754         }
755 
756         return $this->_listenerInstances[$name];
757     }

759 /**
760  * Load a CrudAction instance
761  *
762  * @param string $name The controller action name
763  * @return CrudAction
764  * @throws CakeException If action is not mapped
765  */
766     protected function _loadAction($name) {
767         if (!isset($this->_actionInstances[$name])) {
768             $config = $this->config('actions.' . $name);
769 
770             if (empty($config)) {
771                 throw new CakeException(sprintf('Action "%s" has not been mapped', $name));
772             }
773 
774             list($plugin, $class) = pluginSplit($config['className'], true);
775             $class = ucfirst($class);
776 
777             if (in_array($class, array('Index', 'View', 'Add', 'Edit', 'Delete'))) {
778                 if (!empty($plugin) && $plugin !== 'Crud.') {
779                     throw new CakeException('The build-in CrudActions (Index, View, Add, Edit and Delete) must be loaded from the Crud plugin');
780                 }
781 
782                 $plugin = 'Crud.';
783             }
784 
785             $class .= 'CrudAction';
786             App::uses($class, $plugin . 'Controller/Crud/Action');
787             $subject = $this->getSubject(array('action' => $name));
788             $this->_actionInstances[$name] = new $class($subject, $config);

815 /**
816  * Merge configuration arrays
817  *
818  * Allow us to change e.g. a listener config without losing defaults
819  *
820  * This is like merge_array_recursive - with the difference that
821  * duplicate keys isn't changed to an array with both values, but
822  * overridden
823  *
824  * @param array $array1
825  * @param array $array2
826  * @return array
827  */
828     protected function _mergeConfig(array $array1, array $array2) {
829         $merged = $array1;
830         foreach ($array2 as $key => $value) {
831             if (is_array($value) && isset($merged[$key]) && is_array($merged[$key])) {
832                 $merged[$key] = $this->_mergeConfig($merged[$key], $value);
833                 continue;
834             }
835 
836             $merged[$key] = $value;
837         }
838 
839         return $merged;
840     }
