13 /**
14  * List of components that are capable of dispatching an action that is
15  * not already implement
16  *
17  * @var string
18  */
19     public $dispatchComponents = array();
20 
21 /**
22  * Dispatches the controller action. Checks that the action exists and isn't private.
23  *
24  * If Cake raises MissingActionException we attempt to execute Crud
25  *
26  * @param CakeRequest $request
27  * @return mixed The resulting response.
28  * @throws PrivateActionException When actions are not public or prefixed by _
29  * @throws MissingActionException When actions are not defined and scaffolding and CRUD is not enabled.
30  */
31     public function invokeAction(CakeRequest $request) {
32         try {
33             return parent::invokeAction($request);
34         } catch (MissingActionException $e) {
35             if (!empty($this->dispatchComponents)) {
36                 foreach ($this->dispatchComponents as $component => $enabled) {
37                     if (empty($enabled)) {
38                         continue;
39                     }
40 
41                     // Skip if isActionMapped isn't defined in the Component
42                     if (!method_exists($this->{$component}, 'isActionMapped')) {
43                         continue;
44                     }
45 
46                     // Skip if the action isn't mapped
47                     if (!$this->{$component}->isActionMapped($request->action)) {
48                         continue;
49                     }
50 
51                     // Skip if executeAction isn't defined in the Component
52                     if (!method_exists($this->{$component}, 'executeAction')) {
53                         continue;
54                     }
55 
56                     // Execute the callback, should return CakeResponse object
57                     return $this->{$component}->executeAction();
58                 }
59             }
60 
61             // No additional callbacks, re-throw the normal Cake exception
62             throw $e;
63         }
64     }
65 
66 }
