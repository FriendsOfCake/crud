137 /**
138  * Generic config method
139  *
140  * If $key is an array and $value is empty,
141  * $key will be merged directly with $this->_config
142  *
143  * If $key is a string it will be passed into Hash::insert
144  *
145  * @param mixed $key
146  * @param mixed $value
147  * @return TranslationsEvent
148  */
149     public function config($key = null, $value = null) {
150         if (is_null($key) && is_null($value)) {
151             return $this->_config;
152         }
153 
154         if (empty($value)) {
155             if (is_array($key)) {
156                 $this->_config = Hash::merge($this->_config, $key);
157                 return $this->_config;
158             }
159 
160             return Hash::get($this->_config, $key);
161         }
162 
163         if (is_array($value)) {
164             $merge = Hash::get($this->_config, $key);
165             if ($merge) {
166                 $value += $merge;
167             }
168         }
169 
170         $this->_config = Hash::insert($this->_config, $key, $value);
171         return $this;
172     }

178 /**
179  * SetFlash Crud Event callback
180  *
181  * @throws CakeException if called with invalid args
182  * @param CakeEvent $e
183  * @return void
184  */
185     public function setFlash(CakeEvent $event) {
186         if (empty($event->subject->type)) {
187             throw new CakeException('Missing flash type');
188         }
189 
190         $type = $event->subject->type;
191 
192         $config = Hash::get($this->_config, $type);
193         if (empty($config)) {
194             throw new CakeException('Invalid flash type');
195         }
196 
197         $name = $this->_config['name'] ?: $event->subject->name;
198         $config += array('message' => null, 'element' => null, 'params' => array(), 'key' => 'flash');
199         $message = String::insert($config['message'], array('name' => $name), array('before' => '{', 'after' => '}'));
200 
201         $event->subject->message = __d($this->_config['domain'], $message);
202         $event->subject->element = $config['element'];
203         $event->subject->params = $config['params'];
204         $event->subject->key = $config['key'];
205     }
