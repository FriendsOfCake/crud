 67 /**
 68  * beforeHandle
 69  *
 70  * Called before the crud action is executed
 71  *
 72  * @param CakeEvent $event
 73  * @return void
 74  */
 75     public function beforeHandle(CakeEvent $event) {
 76         parent::beforeHandle($event);
 77 
 78         if (!$this->_request()->is('api')) {
 79             $events = $this->implementedEvents();
 80             $eventManager = $this->_controller()->getEventManager();
 81             foreach (array_keys($events) as $name) {
 82                 if ($name === 'Crud.beforeHandle') {
 83                     continue;
 84                 }
 85                 $eventManager->detach($this, $name);
 86             }
 87             return;
 88         }
 89 
 90         $this->registerExceptionHandler();
 91         $this->_checkRequestMethods();
 92     }

 94 /**
 95  * Check for allowed HTTP request types
 96  *
 97  * @throws BadRequestException
 98  * @return boolean
 99  */
100     protected function _checkRequestMethods() {
101         $action = $this->_action();
102         $apiConfig = $action->config('api');
103 
104         if (!isset($apiConfig['methods'])) {
105             return false;
106         }
107 
108         $request = $this->_request();
109         foreach ($apiConfig['methods'] as $method) {
110             if ($request->is($method)) {
111                 return true;
112             }
113         }
114 
115         throw new BadRequestException('Wrong request method');
116     }

128 /**
129  * Handle response
130  *
131  * @param CakeEvent $event
132  * @return CakeResponse
133  */
134     public function respond(CakeEvent $event) {
135         $subject = $event->subject;
136         $action = $this->_action();
137 
138         $key = $subject->success ? 'success' : 'error';
139         $apiConfig = $action->config('api.' . $key);
140 
141         if (isset($apiConfig['exception'])) {
142             return $this->_exceptionResponse($apiConfig['exception']);
143         }
144 
145         $response = $this->render($event->subject);
146         $response->statusCode($apiConfig['code']);
147         return $response;
148     }

157     protected function _exceptionResponse($exceptionConfig) {
158         $exceptionConfig = array_merge($this->config('exception'), $exceptionConfig);
159 
160         $class = $exceptionConfig['class'];
161 
162         if ($exceptionConfig['type'] === 'validate') {

187 /**
188  * Ensure _serialize is set in the view
189  *
190  * @return void
191  */
192     protected function _ensureSerialize() {
193         $controller = $this->_controller();
194 
195         if (isset($controller->viewVars['_serialize'])) {
196             return;
197         }
198 
199         $action = $this->_action();
200 
201         $serialize = array();
202         $serialize[] = 'success';
203 
204         if (method_exists($action, 'viewVar')) {
205             $serialize['data'] = $action->viewVar();
206         } else {
207             $serialize[] = 'data';
208         }
209 
210         $serialize = array_merge($serialize, (array)$action->config('serialize'));
211         $controller->set('_serialize', $serialize);
212     }

230 /**
231  * Ensure data key is present in Controller:$viewVars
232  *
233  * @param CrudSubject $subject
234  * @return void
235  */
236     protected function _ensureData(CrudSubject $subject) {
237         $controller = $this->_controller();
238 
239         // Don't touch existing data properties
240         if (isset($controller->viewVars['data'])) {
241             return;
242         }
243 
244         $key = $subject->success ? 'success' : 'error';
245 
246         // Load configuration
247         $config = $this->_action()->config('api.' . $key);
248 
249         // New, empty, data array
250         $data = array();
251 
252         // If fields should be extracted from the subject
253         if (isset($config['data']['subject'])) {
254             $config['data']['subject'] = Hash::normalize((array)$config['data']['subject']);
255 
256             $subjectArray = (array)$subject;
257 
258             foreach ($config['data']['subject'] as $keyPath => $valuePath) {
259                 if (is_null($valuePath)) {
260                     $valuePath = $keyPath;
261                 }
262 
263                 $keyPath = $this->_expandPath($subject, $keyPath);
264                 $valuePath = $this->_expandPath($subject, $valuePath);
265 
266                 $data = Hash::insert($data, $keyPath, Hash::get($subjectArray, $valuePath));
267             }
268         }
269 
270         // Raw (hardcoded) key/values
271         if (isset($config['data']['raw'])) {
272 
273             foreach ($config['data']['raw'] as $path => $value) {
274                 $path = $this->_expandPath($subject, $path);
275                 $data = Hash::insert($data, $path, $value);
276             }
277 
278         }
279 
280         // Publish the new data
281         $controller->set('data', $data);
282     }

284 /**
285  * Expand all scalar values from a CrudSubject
286  * and use them for a String::insert() interpolation
287  * of a path
288  *
289  * @param CrudSubject $subject
290  * @param string $path
291  * @return string
292  */
293     protected function _expandPath(CrudSubject $subject, $path) {
294         $keys = array();
295         $subjectArray = (array)$subject;
296 
297         foreach (array_keys($subjectArray) as $key) {
298             if (!is_scalar($subjectArray[$key])) {
299                 continue;
300             }
301 
302             $keys[$key] = $subjectArray[$key];
303         }
304 
305         return String::insert($path, $keys, array('before' => '{', 'after' => '}'));
306     }

355 /**
356  * Setup detectors
357  *
358  * Both detects on two signals:
359  *  1) The extension in the request (e.g. /users/index.$ext)
360  *  2) The accepts header from the client
361  *
362  * There is a combined request detector for all detectors called 'api'
363  *
364  * @return void
365  */
366     public function setupDetectors() {
367         $request = $this->_request();
368         $detectors = $this->config('detectors');
369 
370         foreach ($detectors as $name => $config) {
371 
372             $request->addDetector($name, array('callback' => function(CakeRequest $request) use ($config) {
373                 if (isset($request->params['ext']) && $request->params['ext'] === $config['ext']) {
374                     return true;
375                 }
376 
377                 return $request->accepts($config['accepts']);
378             }));
379 
380         }
381 
382         $request->addDetector('api', array('callback' => function(CakeRequest $request) use ($detectors) {
383             foreach ($detectors as $name => $config) {
384                 if ($request->is($name)) {
385                     return true;
386                 }
387             }
388 
389             return false;
390         }));
391     }

393 /**
394  * Automatically create REST resource routes for all controllers found in your main
395  * application or in a specific plugin to provide access to your resources
396  * using /controller/id.json instead of the default /controller/view/id.json.
397  *
398  * If called with no arguments, all controllers in the main application will be mapped.
399  * If called with a valid plugin name all controllers in that plugin will be mapped.
400  * If combined both controllers from the application and the plugin(s) will be mapped.
401  *
402  * This function needs to be called from your application's app/Config/routes.php:
403  *
404  * ```
405  *     App::uses('ApiListener', 'Crud.Controller/Crud/Listener');
406  *
407  *     ApiListener::mapResources();
408  *     ApiListener::mapResources('DebugKit');
409  *     Router::setExtensions(array('json', 'xml'));
410  *     Router::parseExtensions();
411  * ```
412  *
413  * @static
414  * @param string $plugin
415  * @return void
416  */
417     public static function mapResources($plugin = null) {
418         $key = 'Controller';
419         if ($plugin) {
420             $key = $plugin . '.Controller';
421         }
422 
423         $controllers = array();
424         foreach (App::objects($key) as $controller) {
425             if ($controller !== $plugin . 'AppController') {
426                 if ($plugin) {
427                     $controller = $plugin . '.' . $controller;
428                 }
429 
430                 array_push($controllers, str_replace('Controller', '', $controller));
431             }
432         }
433 
434         Router::mapResources($controllers);
435     }
