103 /**
104  * Redirect callback
105  *
106  * If a special redirect key is provided, change the
107  * redirection URL target
108  *
109  * @param CakeEvent $event
110  * @return void
111  */
112     public function beforeRedirect(CakeEvent $event) {
113         $subject = $event->subject;
114 
115         $redirects = $this->_action()->redirectConfig();
116         if (empty($redirects)) {
117             return;
118         }
119 
120         foreach ($redirects as $redirect) {
121             if (!$this->_getKey($subject, $redirect['reader'], $redirect['key'])) {
122                 continue;
123             }
124 
125             $subject->url = $this->_getUrl($subject, $redirect['url']);
126             break;
127         }
128     }

130 /**
131  * Get the new redirect URL
132  *
133  * Expand configurations where possible and replace the
134  * placeholder with the actual value
135  *
136  * @param CrudSubject $subject
137  * @param array $config
138  * @return array
139  */
140     protected function _getUrl(CrudSubject $subject, array $url) {
141         foreach ($url as $key => $value) {
142             if (!is_array($value)) {
143                 continue;
144             }
145 
146             if ($key === '?') {
147                 $url[$key] = $this->_getUrl($subject, $value);
148                 continue;
149             }
150 
151             $url[$key] = $this->_getKey($subject, $value[0], $value[1]);
152         }
153 
154         return $url;
155     }

157 /**
158  * Return the value of `$type` with `$key`

161  * @param CrudSubject $subject
162  * @param string $reader
163  * @param string $key
164  * @return mixed
165  */
166     protected function _getKey(CrudSubject $subject, $reader, $key) {
167         $callable = $this->reader($reader);
168 
169         if ($callable === null || !is_callable($callable)) {
170             throw new Exception('Invalid reader: ' . $reader);
171         }
172 
173         return $callable($subject, $key);
